	set .@etower_timer2,checkquest(60201,PLAYTIME); // 4 hours

		mes "Faça ou participe de um grupo com mais de um membro e tente novamente.";
		close;

					close;
				}
				mes "^339966[ Pedra de Proteção ]^000000";
				mes "^0000ff"+.@md_name$+"^000000 - Tentar reservar";
				mes "Depois de fazer uma reserva, você tem que falar com o NPC logo atrás e selecionar o menu 'Entrar no Calabouço' para entrar no calabouço.";
				close;
			case 2:
				callsub L_Enter,0,1;
			case 3:
				mes "^339966[ Pedra de Proteção ]^000000";
				mes "Eu irei te levar de volta para Alberta.";
				close2;
				warp "alberta",223,36;
				end;
			case 4:
				close;
			}
		}
		switch(select("^3CB371[»]^000000 Entrar na "+.@md_name$+":^3CB371[»]^000000 Retornar para Alberta:^ff0000[»]^000000 Cancelar")) {
		case 1:
			callsub L_Enter,1,1;
		case 2:
			mes "^339966[ Pedra de Proteção ]^000000";
			mes "Eu irei te levar de volta para Alberta.";
			close2;
			warp "alberta",223,36;
			end;
		case 3:
			end;
		}
	case 0:
	case 1:
		if (.@etower_timer2 < 2 && getcharid(1) == etower_partyid) {
			mes "^339966[ Pedra de Proteção ]^000000";
			mes "Se você já gerou o calabouço, você poderá entrar.";
			next;
			switch(select("^3CB371[»]^000000 Entrar na "+.@md_name$+":^3CB371[»]^000000 Retornar para Alberta:^ff0000[»]^000000 Cancelar")) {
			case 1:
				callsub L_Enter,0,0;
			case 2:
				mes "^339966[ Pedra de Proteção ]^000000";
				mes "Eu irei te levar de volta para Alberta.";
				close2;
				warp "alberta",223,36;
				end;
			case 3:
				close;
			}
		} else {
			set .@dun_lim_time,etower_timer+604800; // 1 week
			// set .@dun_lim_time2,etower_timer+14400; // 4 hours

			set .@dun_ent_t,(.@dun_lim_time - .@dun_cur_time);
